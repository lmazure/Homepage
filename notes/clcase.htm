    #> ct ln -s source target

    #> ct rmname file.c

views
  • list the existing views
    • #> ct lsview
     
  • get information about a view:
    #> ct lsview -long myview

  • create a new view
    • #> ct mkview -tag my_view /export/home/view_store/my_view.vw
     
  • delete an existing view
    • #> ct rmview -tag my_view
     
  • destroy a view that is no more accessible (for example on a dead machine)
    first retrieve the view uuis with
    #> ct lsview -long my_view
    then execute
    #> ct rmview -uuid my_uuid -avobs
    #> ct unregister -view -uuid my_uuid
    #> ct rmtag -view my_view

  • return the name of the setview (the one set by the last "ct setview" command)
    #> ct pwv -setview -short

  • return the name view corresponding to the current directory (it will be different from the setview if with use "cd /view/myview" for exemple)
    #> ct pwv -wdview -short

  • recover a dumped view
    • #> ct reformatview -load -tag my_view
     
  • the active views are accessible in /view
  • start a dynamic view
    #> ct startview my_view
    (mkview, mktag and setview automatically call startview, so this is useful only in situations such as when we want to access the view through the /view path)

labels
  • create a label type
    • #> ct mklbtype MY_LABEL
    -shared to enable to use the label type from any replica
     
  • list all label types
    • #> ct lstype -kind lbtype
     
  • list all the labels sorted by creation date
    #> ct lstype -kind lbtype -fmt "%Nd %n\n" | sort

  • delete a label type and the corresponding labels:
    • #> ct rmtype -rmall lbtype:LM_INTEG

  • label the current version of a file
    • #> ct mklabel MY_LABEL file.c
     
  • move a label on a file
    • #> ct mklabel -replace MY_LABEL file.c@@/main/my_branch/105
     
  • remove a label on a file
    • #> ct rmlabel MY_LABEL fobar.c@@/main/105
     
  • rename a label
    • #> ct rename lbtype:FOO BAR
     
  • extract the files from a list which have / have not a given label
    #> cat list | xargs cleartool describe -fmt "%n %l\n" | grep MY_LABEL | awk '{ print $1 }'

  • find out files with given labels
    • #> ct find . -ver 'version(MY_LABEL_1) || version(MY_LABEL_2)' -print
     
  • check that the latest on the main branch is correctly labeled
    • #> ct find . -ver ' ( version(/main/LATEST) && \!version(MY_LABEL) ) || ( \!version(/main/LATEST) && version(MY_LABEL) ) ' -print

check in/out
  • checkout a file
    • #> ct co foobar.c
    #> ct checkout foobar.c
    -com "comment" with a given comment
    -nc without comment
     
  • uncheckout a file
    • #> ct unco foobar.c
    #> ct unchecout foobar.c
    -keep create a foobar.c.keep backup in the view
    -rm create no backup
     
  • checkin a file
    • #> ct ci foobar.c
    #> ct checkin foobar.c
    -com "comment" with a given comment
    -nc without comment
    -ident force the checkin even if the new version is identical to the previous one
     
  • modify the checkin comment on a file
    • #> ct chevent -replace -c "fix bug XYZge12345" foobar.c@@/main/goto_m3/1

  • append a string to the checkin comment on a file
    • #> ct chevent -c "more information" foobar.c@@/main/goto_m3/1
    or
    #> ct chevent -append -c "more information" foobar.c@@/main/goto_m3/1

  • prepend a string to the checkin comment on a file
    • #> ct chevent -insert -c "some introductional information" foobar.c@@/main/goto_m3/1

  • list the checked out files
    • #> ct lsco -short display only the filename
               -me list only files checked out by current user
               -cview list files check out in current view
               -all lists all the checkouts in the VOB (use the VOB defined by the argument if there is one)
               -fmt "format" lists the informations as defined by the format


  • #> ct lsco -recur dir
    • list files checked out in the filesystem subtree (instead of only the directory level)
    take care the directory dir itself is not  taken into account! (this is the same if no directory name is given, the current directory will not be reported if it is checked out)

  • #> ct lsco -d dir
    • list the checked out version of the directory, not of its elements
     
  • unreserve a checked out file (in a deleted view, for example)
    • #> ct lsco -long

    checkout version "hello.c" from /main/37 (reserved)
    by view: jackson_fix ("oxygen:/usr/home/jackson/ccviews/fix.vws")
    "merge from bugfix branch"
    #> ct unreserve -view oxygen:/usr/home/jackson/ccviews/fix.vws hello.c
     
  • get the checkin comment of a version of a file
    • #> ct describe -fmt '%Nc' foobar.c@@/main/17

    #!/bin/ksh

          echo OK $line

        fi

modification management
  • #> ct diff foo bar
    display the difference between files foo and bar
    ct diff - pred hello.c display the difference between the current version of hello.c and its predecessor
    -serial display in serial format instead of side-by-side
    -graph display the diff in a window

  • check if the files have been modified
    #> foreach file ( `ct lsco -me -short -cview -recur`)
      if { ( ct diff -pred $file > /dev/null ) } then
        echo $file is unchanged
      endif
    end

  • display all the modifications in the checked out files:
    • #> ct lsco -me -short -cview -recur | xargs -n 1 cleartool diff -pred -serial_format
  • desintegrate a day's work
    • 1) get the list of the file and their version
    #> ct find . -version "{ created_by(`whoami`) && created_since(28-Oct-99) && \!created_since(29-Oct-99) && brtype(main) }" -print
    2) apply a negative delta on these files with the following script
    #!/bin/csh
    set file=`echo $1 | sed 's/\(.*\)@@.*/\1/'`
    set version=`echo $1 | sed 's/.*@@\(.*\)/\1/'`
    cleartool checkout-nc $file
    cleartool merge -to $file -delete -version $version

  • #> ct annotate hello.c
    create a file hello.c.ann which contains, for each line, who, when and on which version the line has been changed for the last time.
    -rm also displays the lines that have been removed
    -all also displays all the lines present in any version of the file (i.e. even the lines which are not in a predecessor version, but which are present on other branches)
    -out foobar put the result in the file foobar instead of hello.c.ann
    -out - send the result in stdout instead of hello.c.ann
