pixmap creation from a file (before X11R5)
Pixmap createPixmapFromFile(Screen *screen,
                            const char *bitmapFileName,
                            Pixel foreground,
                            Pixel background,
                            unsigned int *bitmapWidth,
                            unsigned int *bitmapHeight,
                            int *bitmapHotX,
                            int *bitmapHotY,
                            Window window,
                            int depth)
{
    if (window == 0) window = RootWindowOfScreen(screen);
    if (depth == 0) depth = DefaultDepthOfScreen(screen);

    /*** load the bitmap file ***/
    Pixmap bitmap;
    unsigned int width,height;
    int hotX,hotY;

    if ( XReadBitmapFile(DisplayOfScreen(screen),
                         window,
                         bitmapFileName,
                         &width,
                         &height,
                         &bitmap,
                         &hotX,
                         &hotY) != BitmapSuccess ) {
         (void)fprintf(stderr,
                       "Cannot create pixmap from file %s\n",
                       bitmapFileName);
         return(0);}
 

    /*** create the GC ***/
    XGCValues gcValues;
    gcValues.foreground = foreground;
    gcValues.background = background;
    unsigned long gcMask = GCForeground | GCBackground;
    GC gc=XCreateGC(DisplayOfScreen(screen),
                    window,
                    gcMask,
                    &gcValues);

    /*** create the pixmap ***/
    Pixmap pixmap=XCreatePixmap(DisplayOfScreen(screen),
                                window,
                                width,
                                height,
                                depth);
    XCopyPlane(DisplayOfScreen(screen),
               bitmap,
               pixmap,
               gc,
               0,0,
               width,height,
               0,0,
               1);

    /*** free temporary resources ***/
    XFreePixmap(DisplayOfScreen(screen),
                bitmap);
    XFreeGC(DisplayOfScreen(screen),
            gc);

    /*** return pixmap geometry if required ***/
    if (bitmapWidth) *bitmapWidth=width;
    if (bitmapHeight) *bitmapHeight=height;
    if (bitmapHotX) *bitmapHotX=hotX;
    if (bitmapHotY) *bitmapHotY=hotY;

    return(pixmap);
}

pixmap creation from a file (with X11R5 or later)
Pixmap createPixmapFromFile(Screen *screen,
                            const char *bitmapFileName,
                            Pixel foreground,
                            Pixel background,
                            unsigned int *bitmapWidth,
                            unsigned int *bitmapHeight,
                            int *bitmapHotX,
                            int *bitmapHotY,
                            Window window,
                            int depth)
{
    if (window == 0) window = RootWindowOfScreen(screen);
    if (depth == 0) depth = DefaultDepthOfScreen(screen);

    /*** load the bitmap file ***/
    unsigned int width,height;
    int hotX,hotY;
    unsigned char *data;

    if ( XReadBitmapFileData(bitmapFileName,
                             &width,
                             &height,
                             &data,
                             &hotX,
                             &hotY) != BitmapSuccess ) {
       (void)fprintf(stderr,
                     "Cannot load pixamp from file %s\n",
                     bitmapFileName);
       return(0);}
 

    /*** create the pixmap ***/
    Pixmap pixmap=XCreatePixmapFromBitmapData(DisplayOfScreen(screen),
                                              window,
                                              (char*)data,
                                              width,
                                              height,
                                              foreground,
                                              background,
                                              depth);

    /*** free temporary memory ***/
    XFree(data);

    /*** return pixmap geometry if required ***/
    if (bitmapWidth) *bitmapWidth=width;
    if (bitmapHeight) *bitmapHeight=height;
    if (bitmapHotX) *bitmapHotX=hotX;
    if (bitmapHotY) *bitmapHotY=hotY;

    return(pixmap);
}

create a pixmap in a safe way (avoir a BadAlloc error is there is not enough memory)
static bool XSafeCreatePixmap_error;

static int pixmapErrorHandler(Display *dpy, XErrorEvent *error)
{
    XSafeCreatePixmap_error = True;
}

Pixmap XSafeCreatePixmap(Display *dpy,
                         Drawable drawable,
                         unsigned width,
                         unsigned height,
                         unsigned depth)
{
    /* synchronise to avoid possible previous error */ 
    XSync(display,False); 

    /* initialize the error flag */
    XSafeCreatePixmap_error = False;

    /* install our error handler */
    XErrorHandler oldHandler = XSetErrorHandler(pixmapErrorHandler);

    /* attempt to create pixmap */
    Pixmap thePixmap = XCreatePixmap(dpy, drawable, width, height, depth);

    /* do we really need this? I don't think so
    Drawable root;
    int x, y;
    unsigned int rw, rh,bw,rd;
    XGetGeometry(dpy, thePixmap, &root, &x, &y, &rw, &rh, &bw, &rd);
    */

    /* wait to see if server allocated pixmap */
    XSync(dpy, False);

    /* restore previous error handler */
    XSetErrorHandler(oldHandler);

    return  XSafeCreatePixmap_error ? (Pixmap)NULL : thePixmap;
}
