}

getting a visual and a colormap
  int theDepth = 8;

  // -------------------------------------------------------------

  Visual *visual;
  XVisualInfo request;
  XVisualInfo* visualList;
  int sizeVisualList;

  request.screen=0;
  request.depth=theDepth;
  request.c_class=PseudoColor;

  visualList=XGetVisualInfo(dpy,
                            VisualScreenMask|VisualDepthMask|VisualClassMask,
                            &request,
                            &sizeVisualList);
  if (sizeVisualList==0)
  {
    fprintf(stderr, "Warning: failed to get visual on screen 0 for depth %d\n",theDepth);
    return(EXIT_FAILURE);
  }

  visual = visualList[0].visual;
  XFree(visualList);

  // -------------------------------------------------------------

  Colormap cmap;

  cmap=XCreateColormap(dpy,
                       RootWindow(dpy,0),
                       visual,
                       AllocNone);

  // -------------------------------------------------------------

  Widget shell = XtVaAppCreateShell("shell",
                                    "Shell",
                                    applicationShellWidgetClass,
                                    dpy,
                                    XmNdepth, theDepth,
                                    XmNvisual, visual,
                                    XmNcolormap, cmap,
                                    NULL);

create a shell on the screen defined by a resource
const char *shellName = "remoteBrowser";
const char *shellClass = "RemoteBrowser";

int screenNumber;
XrmValue val;
char *resType;
XrmDatabase xrdb = XtDatabase(display);

char *n = new char[strlen(shellName)+8];
char *c = new char[strlen(shellClass)+8];
sprintf(n,"%s.%s",shellName,"screen");
sprintf(c,"%s.%s",shellClass,"screen");

if (XrmGetResource(xrdb, n, c, &resType, &val))
{
    if(sscanf(val.addr, "%d", &screenNumber)!=1 ||
       screenNumber<0 ||
       screenNumber>=XScreenCount(display) )
    {
         std::cerr << "bad screen resource (\"" << val.addr << "\") for " << shellName;
         screenNumber = XDefaultScreen(display);
    }
}
else
{
     std::cerr << "cannot retrieve screen resource for " << shellName;
     screenNumber = XDefaultScreen(display);
}

delete[] n;
delete[] c;

Screen *scr = XScreenOfDisplay(display,screenNumber);
shell = XtVaAppCreateShell(shellName,
                           shellClass,
                           topLevelShellWidgetClass,
                           display,
                           XmNscreen, scr,
                           NULL);
